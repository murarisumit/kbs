Parsing flowlogs

parse @message '* * * * * * * * * * * * * * * * * *' as  version, account_id, interface_id, srcaddr, dstaddr, srcport, dstport, protocol, packets, bytes, action, log_status, tcp_flags, pkt_srcaddr, pkt_dstaddr, pkt_src_aws_service, pkt_dst_aws_service, flow_direction
| stats sum(bytes) as bytesTransferred by pkt-src-aws-service pkt-dst-aws-service, az
| sort bytesTransferred desc
| limit 30

parse @message '* * * * * * * * * * * * * * * * * * * * * * * * * *' as action,az,bytes,dstAddr,dstPort, end, flow, instanceId, ifaceId, logstatus, pkts, dstAws, pktDstAddr, srcAws, pktSrcAddr, protocol, region, srcAddr, srcPort, start, sublocId, sublocType, subnetId, tcpFlags, tpath, type
| filter (isIpv4InSubnet(srcAddr, "192.168.2.0/24") or isIpv4InSubnet(srcAddr, "192.168.3.0/24") or isIpv4InSubnet(srcAddr, "192.168.32.0/20") or isIpv4InSubnet(srcAddr, "192.168.0.0/24") or isIpv4InSubnet(srcAddr, "192.168.1.0/24") or isIpv4InSubnet(srcAddr, "192.168.16.0/20") ) and (isIpv4InSubnet(dstAddr, "192.168.4.0/24") or isIpv4InSubnet(dstAddr, "192.168.5.0/24") or isIpv4InSubnet(dstAddr, "192.168.48.0/24"))
| stats sum(bytes) as bytesTransferred by dstAddr, srcAddr, az, subnetId, flow, type
| sort bytesTransferred desc
| limit 25
