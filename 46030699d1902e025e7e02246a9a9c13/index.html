ls -f | xargs -I {} /bin/bash -c 'echo test1-{}; echo test2-{}'
ls -f | xargs -I % /bin/bash -c 'echo test1-%; echo test2-%'
ls -f | xargs -i -pt echo "testxargs-{}"
